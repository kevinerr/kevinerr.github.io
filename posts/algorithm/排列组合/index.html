public:
    vector<vector<int>> res;
    vector<int> cur;
    void dfs(int start,int k, int target,int sum){
        if(target==sum&&k==0){
            res.push_back(cur);
            return;
        }       
        if(sum>target) return;
        for(int i =start;i<=9;i++){
            sum+=i;
            cur.push_back(i);
            k--;
            dfs(i+1,k,target,sum);
            sum-=i;
            cur.pop_back();
            k++;
        }
    }
    vector<vector<int>> combinationSum3(int k, int n) {
        dfs(1,k,n,0);
        return res;
    }
};
